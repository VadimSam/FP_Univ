%prompt = 'What is matrix A?\n>>';
%A = input(prompt);
Asz=size(A);
%prompt = 'What is matrix B?\n>>';
%B = input(prompt);
Bsz=size(B);
if(Asz(1,1)~=Bsz(1,2))
disp("Can't proceed,due to n of A rows does't equal to n of B cols!");
return;
end
fprintf("Original matrices\n");
display(A);
display(B);
fprintf("Matrices size\n");
fprintf("A\t%ix%i\nB\t%ix%i\n",Asz(1,1),Asz(1,2),Bsz(1,1),Bsz(1,2));
fprintf("Transpose matrices\n");
Atrpd = A';
Btrpd = B';
fprintf("A_{T}\n");
display(Atrpd);
fprintf("B_{T}\n");
display(Btrpd);
fprintf("Matrices determinants\n");
Adet = det(A);
Bdet = det(B);
fprintf("∆A = %d\n∆B = %d\n", Adet,Bdet);
fprintf("Adjoint matrices\n");
Aadj = adjoint(A);
Badj = adjoint(B);
fprintf("A*\n");
display(Aadj);
fprintf("B*\n");
display(Badj);
fprintf("Inverted matrices\nA^{-1}");
Ainv = pinv(A);
display(Ainv);
fprintf("B^{-1}\n");
Binv = pinv(B);
display(Binv);
fprintf("Matrix multiplication\n");
Cab = A*B;
Cba = B*A;
fprintf("A*B=Cab\n");
display(Cab);
fprintf("B*A=Cba\n");
display(Cba);
fprintf("Matrix rank\n");
Arank = rank(A);
Brank = rank(B);
fprintf("r(A) = %i\nr(B) = %i\n",Arank, Brank);
fprintf("\nC.F. Gauss method\n");
Arref = rref(A);
Brref = rref(B);
fprintf("Row echelons\n");
display(Arref);
display(Brref);
X = Arref/Brref;
fprintf("System solution\n");
disp(X);
